function languageCss(str){
	return(str)	//comments
	.replace(/\/\/(.*?)\n/ , '$&')
	//class
	.replace(/\.(.*?)[a-z|A-Z|0-9]{0,}/g , '$&')
	//id
	.replace(/#(.*?){/g , '#$1{')
	//hex number
	.replace(/#(.*?);/g , '#$1;')
	//number

	//px
	.replace(/px/,'$&')
	//tags
	.replace(/\n(.*?){/g, '\n$1{')
}
